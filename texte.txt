function formatMessage(text) {
  text = text
    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    .replace(/```(\w+)?\n([\s\S]*?)```/g, '<pre><code>$2</code></pre>')
    .replace(/`([^`]+)`/g, '<code>$1</code>')
    .replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
    .replace(/\*([^*]+)\*/g, '<em>$1</em>')
    .replace(/^### (.+)$/gm, '<h3>$1</h3>')
    .replace(/^## (.+)$/gm, '<h2>$1</h2>')
    .replace(/^# (.+)$/gm, '<h1>$1</h1>')
    .replace(/^- (.+)$/gm, '<li>$1</li>')
    .replace(/(<li>.*<\/li>)/gs, '<ul>$1</ul>')
    .replace(/\n\n/g, '</p><p>')
    .replace(/\n/g, '<br>');
  return '<p>' + text + '</p>';
}

function addMessage(role, content, isError = false) {
  const welcome = document.getElementById('welcome');
  if (welcome) welcome.remove();
  const chat = document.getElementById('chat');
  const msgDiv = document.createElement('div');
  msgDiv.className = `message ${role}`;
  const avatarDiv = document.createElement('div');
  avatarDiv.className = `avatar ${role}-avatar`;
  avatarDiv.textContent = role === 'ai' ? '✦' : '👤';
  const bubbleDiv = document.createElement('div');
  bubbleDiv.className = `bubble${isError ? ' error-bubble' : ''}`;
  bubbleDiv.innerHTML = role === 'ai'
    ? formatMessage(content)
    : content.replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/\n/g,'<br>');
  msgDiv.appendChild(avatarDiv); msgDiv.appendChild(bubbleDiv);
  chat.appendChild(msgDiv);
  chat.scrollTop = chat.scrollHeight;
  return bubbleDiv;
}

function addTyping() {

  if (welcome) welcome.remove();

async function sendMessage() {
  if (isLoading) return;
  const input = document.getElementById('userInput');
  const text = input.value.trim();
  if (!text) return;
  input.value = ''; input.style.height = 'auto';
  isLoading = true;
  document.getElementById('sendBtn').disabled = true;
  addMessage('user', text);
  messages.push({ role: 'user', content: text });
  const typingEl = addTyping();
  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': API_KEY,
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: SYSTEM_PROMPT,
        messages: messages
      })
    });
    const data = await response.json();
    typingEl.remove();
    if (data.error) {
      addMessage('ai', '⚠️ Erreur : ' + data.error.message, true);
    } else {
      const reply = data.content.map(b => b.type === 'text' ? b.text : '').join('');
      addMessage('ai', reply);
      messages.push({ role: 'assistant', content: reply });
    }
  } catch (err) {
    typingEl.remove();
    addMessage('ai', '⚠️ Impossible de contacter le serveur. Vérifiez votre connexion.', true);
  }
  isLoading = false;
  document.getElementById('sendBtn').disabled = false;
  document.getElementById('userInput').focus();
}
